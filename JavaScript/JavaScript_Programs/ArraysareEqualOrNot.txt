const points = [1, 1, 4, 3, 2, 1];
const points1 = [1, 1, 3, 4, 2, 1];

// Step 1:  Check Length Equality
//          a.length === b.length ensures the two arrays are the same size.

//          If lengths differ, the arrays cannot be equal, so the function returns false immediately.

// Step 2:  Element-wise Comparison using .every()
//          If lengths are equal, a.every((val, i) => val === b[i]) is called.
//          .every() checks if every element in a passes the test given by the callback function.
//          The callback compares each element val in a with the corresponding element in b at the same index i.
//          If any element pair is different, .every() returns false and so does arraysEqual.
//          If all elements match exactly, .every() returns true and so does arraysEqual.
function max(arr1, arr2) {
    return arr1.length === arr2.length && arr1.every((val, i) => val === arr2[i]);
}

console.log(max(points, points1))
